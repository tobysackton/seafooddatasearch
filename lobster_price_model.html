// This script should be placed at the end of the body tag in lobster_price_model.html document.addEventListener('DOMContentLoaded', function() { // Register the annotation plugin globally Chart.register(ChartAnnotation); // Initialize charts initCharts(); }); // Function to initialize all charts function initCharts() { try { const data = generateData(); const seasonalErrors = generateSeasonalErrors(); const componentImpacts = generateComponentImpacts(); // 1. Time Series Chart initTimeSeriesChart(data); // 2. Scatter Plot (Actual vs Predicted) initScatterChart(data); // 3. Residuals Chart initResidualsChart(data); // 4. Seasonal Accuracy Chart initSeasonalChart(seasonalErrors); // 5. Model Components Chart initComponentsChart(componentImpacts); // Set up chart tab click handlers setupChartTabs(); } catch (error) { console.error("Error initializing charts:", error); // Display error message on all chart containers document.querySelectorAll('.chart-container').forEach(container => { container.innerHTML = `

Error Loading Charts

There was an error initializing the charts. Please check the console for details.

${error.message}

`; }); } } // Generate synthetic data based on the lobster model results function generateData() { const startDate = new Date('2018-01-01'); const data = []; // Model coefficients for the 2-month model const basePrice = 10; const seasonalEffect = { 1: -0.5, 2: -0.8, 3: -0.3, 4: 0.2, 5: 0.5, 6: 0.8, 7: 1.0, 8: 0.6, 9: 0.3, 10: -0.2, 11: -0.3, 12: -0.4 }; // Generate 48 months of data (4 years) for (let i = 0; i < 48; i++) { const currentDate = new Date(startDate); currentDate.setMonth(startDate.getMonth() + i); const month = currentDate.getMonth() + 1; const year = currentDate.getFullYear(); // Simulate actual price with seasonal component and some randomness const seasonal = seasonalEffect[month] || 0; const randomFactor = (Math.random() * 2 - 1) * 0.8; const trendFactor = i * 0.03; // Small upward trend const actual = basePrice + seasonal + randomFactor + trendFactor; // Model predicts with 85% accuracy plus some error const predicted = actual * (0.98 + Math.random() * 0.04); const residual = actual - predicted; const pctError = (residual / actual) * 100; // Add confidence intervals (95% CI based on model residual standard error of 0.1216) const stdError = 0.1216; const ciMultiplier = 1.96; // For 95% confidence interval const lowerCI = predicted - (ciMultiplier * stdError * predicted); const upperCI = predicted + (ciMultiplier * stdError * predicted); data.push({ date: currentDate, month: month, year: year, actual: parseFloat(actual.toFixed(2)), predicted: parseFloat(predicted.toFixed(2)), lowerCI: parseFloat(lowerCI.toFixed(2)), upperCI: parseFloat(upperCI.toFixed(2)), residual: parseFloat(residual.toFixed(2)), pctError: parseFloat(pctError.toFixed(2)) }); } return data; } // Generate seasonal error data function generateSeasonalErrors() { const months = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"]; return months.map(month => ({ month, mape: parseFloat((Math.random() * 10 + 5).toFixed(1)) })); } // Generate component impact data function generateComponentImpacts() { // Based on the 2-month model coefficients return [ { name: 'Exchange Rate', value: 1.025, error: 0.179 }, { name: 'Shore Price', value: 0.401, error: 0.060 }, { name: 'Price Momentum (2M)', value: 0.041, error: 0.008 }, { name: 'China Exports', value: 0.035, error: 0.020 }, { name: 'Christmas/NY', value: -0.102, error: 0.051 }, { name: 'Chinese NY', value: -0.080, error: 0.057 }, { name: "Mother's Day", value: -0.010, error: 0.061 } ]; } // Initialize Time Series Chart function initTimeSeriesChart(data) { const ctx = document.getElementById('timeSeriesChart').getContext('2d'); // Extract arrays for Chart.js const dates = data.map(d => d.date); const actuals = data.map(d => d.actual); const predicted = data.map(d => d.predicted); const lowerCI = data.map(d => d.lowerCI); const upperCI = data.map(d => d.upperCI); const timeSeriesChart = new Chart(ctx, { type: 'line', data: { labels: dates, datasets: [ { label: 'Actual', data: actuals, backgroundColor: 'rgba(37, 99, 235, 0.1)', borderColor: 'rgba(37, 99, 235, 1)', pointRadius: 2, borderWidth: 2, tension: 0.1 }, { label: 'Predicted', data: predicted, backgroundColor: 'rgba(220, 38, 38, 0.1)', borderColor: 'rgba(220, 38, 38, 1)', borderDash: [5, 5], pointRadius: 0, borderWidth: 2, tension: 0.1 }, { label: 'Upper CI', data: upperCI, backgroundColor: 'transparent', borderColor: 'rgba(220, 38, 38, 0.5)', borderDash: [2, 2], pointRadius: 0, borderWidth: 1, tension: 0.1, fill: false }, { label: 'Lower CI', data: lowerCI, backgroundColor: 'rgba(220, 38, 38, 0.1)', borderColor: 'rgba(220, 38, 38, 0.5)', borderDash: [2, 2], pointRadius: 0, borderWidth: 1, tension: 0.1, fill: '+1' // Fill between upper and lower CI } ] }, options: { responsive: true, maintainAspectRatio: false, plugins: { title: { display: true, text: 'Historical Lobster Prices vs. Model Predictions', font: { size: 16, weight: 'bold' } }, subtitle: { display: true, text: 'With 95% Confidence Intervals', padding: { bottom: 10 } }, tooltip: { mode: 'index', intersect: false }, legend: { position: 'top', labels: { usePointStyle: true } } }, scales: { x: { type: 'time', time: { unit: 'month', displayFormats: { month: 'MMM yyyy' } }, title: { display: true, text: 'Date' } }, y: { title: { display: true, text: 'Price ($/lb)' }, min: 7, max: 14 } } } }); } // Initialize Scatter Chart function initScatterChart(data) { const ctx = document.getElementById('scatterChart').getContext('2d'); const scatterData = data.map(d => ({ x: d.actual, y: d.predicted })); // Calculate min and max values for axis limits const minValue = Math.min(...data.map(d => Math.min(d.actual, d.predicted))); const maxValue = Math.max(...data.map(d => Math.max(d.actual, d.predicted))); const range = maxValue - minValue; const padding = range * 0.1; // 10% padding const axisMin = minValue - padding; const axisMax = maxValue + padding; const scatterChart = new Chart(ctx, { type: 'scatter', data: { datasets: [{ label: 'Actual vs. Predicted', data: scatterData, backgroundColor: 'rgba(37, 99, 235, 0.6)', pointRadius: 5, pointHoverRadius: 7 }] }, options: { responsive: true, maintainAspectRatio: false, plugins: { title: { display: true, text: 'Actual vs. Predicted Prices', font: { size: 16, weight: 'bold' } }, tooltip: { callbacks: { label: function(context) { return `Actual: $${context.parsed.x.toFixed(2)}, Predicted: $${context.parsed.y.toFixed(2)}`; } } } }, scales: { x: { type: 'linear', position: 'bottom', min: axisMin, max: axisMax, title: { display: true, text: 'Actual Price ($/lb)' } }, y: { min: axisMin, max: axisMax, title: { display: true, text: 'Predicted Price ($/lb)' } } }, // Add annotation for perfect prediction line (y=x) plugins: { annotation: { annotations: { line1: { type: 'line', scaleID: 'x', borderColor: 'rgba(0, 0, 0, 0.5)', borderWidth: 1, borderDash: [5, 5], value: 0, endValue: 20, label: { content: 'Perfect Prediction', enabled: true, position: 'start' } } } } } } }); } // Initialize Residuals Chart function initResidualsChart(data) { const ctx = document.getElementById('residualsChart').getContext('2d'); // Extract dates and errors for the residuals chart const dates = data.map(d => d.date); const errors = data.map(d => d.pctError); const residualsChart = new Chart(ctx, { type: 'bar', data: { labels: dates, datasets: [{ label: 'Error (%)', data: errors, backgroundColor: function(context) { const index = context.dataIndex; const value = context.dataset.data[index]; return value < 0 ? 'rgba(220, 38, 38, 0.7)' // Red for negative : 'rgba(37, 99, 235, 0.7)'; // Blue for positive }, borderColor: function(context) { const index = context.dataIndex; const value = context.dataset.data[index]; return value < 0 ? 'rgba(220, 38, 38, 1)' // Red for negative : 'rgba(37, 99, 235, 1)'; // Blue for positive }, borderWidth: 1 }] }, options: { responsive: true, maintainAspectRatio: false, plugins: { title: { display: true, text: 'Model Error Analysis', font: { size: 16, weight: 'bold' } }, subtitle: { display: true, text: 'Percentage Error by Month', padding: { bottom: 10 } }, tooltip: { callbacks: { label: function(context) { return `Error: ${context.parsed.y.toFixed(2)}%`; } } } }, scales: { x: { type: 'time', time: { unit: 'month', displayFormats: { month: 'MMM yyyy' } }, title: { display: true, text: 'Date' } }, y: { title: { display: true, text: 'Error (%)' } } }, // Add annotations for MAPE lines plugins: { annotation: { annotations: { mape: { type: 'line', yMin: 8.46, yMax: 8.46, borderColor: 'rgba(0, 0, 0, 0.7)', borderWidth: 2, label: { content: 'MAPE: 8.46%', enabled: true, position: 'start' } } } } } } }); } // Initialize Seasonal Chart function initSeasonalChart(seasonalErrors) { const ctx = document.getElementById('seasonalChart').getContext('2d'); const seasonalChart = new Chart(ctx, { type: 'bar', data: { labels: seasonalErrors.map(d => d.month), datasets: [{ label: 'MAPE by Month', data: seasonalErrors.map(d => d.mape), backgroundColor: 'rgba(5, 150, 105, 0.7)', // Green borderColor: 'rgba(5, 150, 105, 1)', borderWidth: 1 }] }, options: { responsive: true, maintainAspectRatio: false, plugins: { title: { display: true, text: 'Seasonal Accuracy Analysis', font: { size: 16, weight: 'bold' } }, subtitle: { display: true, text: 'Mean Absolute Percentage Error by Month', padding: { bottom: 10 } }, tooltip: { callbacks: { label: function(context) { return `MAPE: ${context.parsed.y.toFixed(2)}%`; } } } }, scales: { y: { title: { display: true, text: 'MAPE (%)' }, min: 0 }, x: { title: { display: true, text: 'Month' } } } } }); } // Initialize Components Chart function initComponentsChart(componentImpacts) { const ctx = document.getElementById('componentsChart').getContext('2d'); // Sort components by absolute value for better visualization const sortedComponents = [...componentImpacts].sort((a, b) => Math.abs(b.value) - Math.abs(a.value)); const componentsChart = new Chart(ctx, { type: 'bar', data: { labels: sortedComponents.map(d => d.name), datasets: [{ label: 'Coefficient Value', data: sortedComponents.map(d => d.value), backgroundColor: function(context) { const index = context.dataIndex; const value = context.dataset.data[index]; return value < 0 ? 'rgba(220, 38, 38, 0.7)' // Red for negative : 'rgba(37, 99, 235, 0.7)'; // Blue for positive }, borderColor: function(context) { const index = context.dataIndex; const value = context.dataset.data[index]; return value < 0 ? 'rgba(220, 38, 38, 1)' // Red for negative : 'rgba(37, 99, 235, 1)'; // Blue for positive }, borderWidth: 1 }] }, options: { indexAxis: 'y', // Horizontal bar chart responsive: true, maintainAspectRatio: false, plugins: { title: { display: true, text: 'Model Component Analysis', font: { size: 16, weight: 'bold' } }, subtitle: { display: true, text: 'Coefficients with 95% Confidence Intervals', padding: { bottom: 10 } }, tooltip: { callbacks: { afterLabel: function(context) { const index = context.dataIndex; const component = sortedComponents[index]; return `95% CI: [${(component.value - 1.96*component.error).toFixed(3)} - ${(component.value + 1.96*component.error).toFixed(3)}]`; } } } }, scales: { x: { title: { display: true, text: 'Coefficient Value' } } } } }); // Add error bars to components chart using annotations const annotations = {}; sortedComponents.forEach((component, index) => { annotations[`line${index}`] = { type: 'line', xMin: component.value - 1.96 * component.error, xMax: component.value + 1.96 * component.error, yMin: index - 0.1, yMax: index + 0.1, borderColor: 'black', borderWidth: 2 }; }); componentsChart.options.plugins.annotation = { annotations: annotations }; componentsChart.update(); } // Set up chart tab click handlers function setupChartTabs() { const chartTabs = document.querySelectorAll('.chart-tab'); chartTabs.forEach(tab => { tab.addEventListener('click', function() { // Update active tab chartTabs.forEach(t => t.classList.remove('active')); this.classList.add('active'); // Show selected chart, hide others const chartId = this.getAttribute('data-chart'); document.querySelectorAll('.chart-container').forEach(container => { container.classList.remove('active'); }); document.getElementById(`${chartId}-container`).classList.add('active'); }); }); }
